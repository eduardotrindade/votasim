CREATE DATABASE sistema_eventos;
USE sistema_eventos;

-- PAPEL
CREATE TABLE papel (
    id INT AUTO_INCREMENT PRIMARY KEY,
    papel VARCHAR(50)
);

-- UF
CREATE TABLE uf (
    id INT AUTO_INCREMENT PRIMARY KEY,
    sigla CHAR(2),
    nome VARCHAR(100)
);

-- CIDADE
CREATE TABLE cidade (
    id INT AUTO_INCREMENT PRIMARY KEY,
    uf_id INT,
    nome VARCHAR(100),
    FOREIGN KEY (uf_id) REFERENCES uf(id)
);

-- REGIAO
CREATE TABLE regiao (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cidade_id INT,
    nome VARCHAR(100),
    FOREIGN KEY (cidade_id) REFERENCES cidade(id)
);

-- AGRUPAMENTO
CREATE TABLE agrupamento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    pessoa_referencia_id INT,
    nome VARCHAR(100),
    regiao_id INT,
    endereco VARCHAR(255),
    cep VARCHAR(20),
    observacao TEXT,
    FOREIGN KEY (regiao_id) REFERENCES regiao(id)
);

-- PESSOA
CREATE TABLE pessoa (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(150),
    data_nascimento DATE,
    papel_id INT,
    foto TEXT,
    observacao TEXT,
    anexos BLOB,
    agrupamento_id INT,
    cadastrado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    usuario_id INT,
    telefone VARCHAR(20),
    email VARCHAR(100),
    FOREIGN KEY (papel_id) REFERENCES papel(id),
    FOREIGN KEY (agrupamento_id) REFERENCES agrupamento(id)
);

-- EVENTO
CREATE TABLE evento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    regiao_id INT,
    data DATE,
    observacao TEXT,
    anexo BLOB,
    usuario_organizador_id INT,
    titulo VARCHAR(150),
    check_in BOOLEAN,
    data_hora_check_in DATETIME,
    localizacao_check_in VARCHAR(255),
    latitude DOUBLE,
    longitude DOUBLE,
    FOREIGN KEY (regiao_id) REFERENCES regiao(id)
);

-- EVENTO_PESSOAS
CREATE TABLE evento_pessoas (
    evento_id INT,
    pessoa_id INT,
    convidada BOOLEAN,
    presente BOOLEAN,
    PRIMARY KEY (evento_id, pessoa_id),
    FOREIGN KEY (evento_id) REFERENCES evento(id),
    FOREIGN KEY (pessoa_id) REFERENCES pessoa(id)
);

-- EVENTO_FOTOS
CREATE TABLE evento_fotos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    evento_id INT,
    foto TEXT,
    cadastrado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    localizacao_foto VARCHAR(255),
    FOREIGN KEY (evento_id) REFERENCES evento(id)
);

-- VINCULO
CREATE TABLE vinculo (
    id INT AUTO_INCREMENT PRIMARY KEY,
    candidato_id INT,
    assessor_id INT,
    lider_id INT,
    pessoa_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    esta_ativo BOOLEAN
);

-- REGISTRO ESPONTANEO
CREATE TABLE registro_espontaneo (
    id INT AUTO_INCREMENT PRIMARY KEY,
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    latitude DOUBLE,
    longitude DOUBLE,
    foto TEXT,
    usuario_id INT,
    observacao TEXT
);

🛡️ Middleware de Permissão
function authorize(roles = []) {
  return (req, res, next) => {
    const user = req.user;
    if (!roles.includes(user.role)) {
      return res.status(403).send('Acesso negado');
    }
    next();
  };
}

import { GoogleMap, Marker, useLoadScript } from '@react-google-maps/api';
import { useEffect, useState } from 'react';
import api from '../services/api';

export default function MapaEventos() {
  const { isLoaded } = useLoadScript({
    googleMapsApiKey: import.meta.env.VITE_GOOGLE_MAPS_API_KEY
  });

  const [eventos, setEventos] = useState([]);

  useEffect(() => {
    api.get('/eventos').then(res => setEventos(res.data));
  }, []);

  if (!isLoaded) return <div>Carregando mapa...</div>;

  return (
    <GoogleMap
      zoom={5}
      center={{ lat: -15.78, lng: -47.93 }}
      mapContainerStyle={{ width: '100%', height: '500px' }}
    >
      {eventos.map(e => (
        <Marker
          key={e.id}
          position={{ lat: e.latitude, lng: e.longitude }}
        />
      ))}
    </GoogleMap>
  );
}

📄 Capturar localização (frontend)
function CheckIn({ eventoId }) {

  const fazerCheckin = () => {
    navigator.geolocation.getCurrentPosition((pos) => {
      const { latitude, longitude } = pos.coords;

      api.post(`/eventos/${eventoId}/checkin`, {
        latitude,
        longitude
      });
    });
  };

  return <button onClick={fazerCheckin}>Fazer Check-in</button>;
}

function calcularDistancia(lat1, lon1, lat2, lon2) {
  const R = 6371;
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;

  const a =
    Math.sin(dLat/2) * Math.sin(dLat/2) +
    Math.cos(lat1 * Math.PI / 180) *
    Math.cos(lat2 * Math.PI / 180) *
    Math.sin(dLon/2) * Math.sin(dLon/2);

  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}

router.get('/bi', async (req, res) => {

  const [eventos] = await db.execute(`
    SELECT COUNT(*) as total FROM evento
  `);

  const [pessoas] = await db.execute(`
    SELECT COUNT(*) as total FROM pessoa
  `);

  const [presencas] = await db.execute(`
    SELECT COUNT(*) as total FROM evento_pessoas WHERE presente = true
  `);

  const [porRegiao] = await db.execute(`
    SELECT r.nome, COUNT(e.id) as total
    FROM evento e
    JOIN regiao r ON e.regiao_id = r.id
    GROUP BY r.nome
  `);

  res.json({
    eventos: eventos[0].total,
    pessoas: pessoas[0].total,
    presencas: presencas[0].total,
    porRegiao
  });
});
